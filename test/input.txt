void print(int i);
void printd(double v);

int factorial(int n) {
    if (n <= 1) return 1;
    else return n*factorial(n-1);
}

void printAll(int max) {
    int x = 0;
    while(x < max) {
        int f = factorial(x);
        print(f);
        x = x+1;
    }
}
